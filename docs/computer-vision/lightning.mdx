```python copy
class MLPClassifierLight(pl.LightningModule): # ë¶€ëª¨ í´ë˜ìŠ¤ê°€ ë°”ë€œ
    def __init__(self, input_size=784):
        super().__init__()
        self.linear1 = nn.Linear(input_size, 16)  # 16ê°œì˜ ë‰´ëŸ°ì„ ê°€ì§„ ì€ë‹‰ì¸µ
        self.linear2 = nn.Linear(16, 1)  # í•œ ê°œì˜ ì¶œë ¥ ë‰´ëŸ°
        self.accuracy = BinaryAccuracy()

    def forward(self, x):
        x = x.view(-1, 784) # Flatten ë ˆì´ì–´ ì—†ì´, í…ì„œ ëª¨ì–‘ì„ (N, 28, 28) â†’ (N, 784)
        x = F.relu(self.linear1(x)) # ReLU í™œì„±í™” í•¨ìˆ˜
        x = F.sigmoid(self.linear2(x))  # í™œì„±í™” í•¨ìˆ˜ëŠ” __init__ì—ì„œ ì •ì˜ X
        return x

    def process_batch(self, batch):   # ë°°ì¹˜ ì²˜ë¦¬
        images, labels = batch
        outputs = self(images).squeeze()
        loss = F.binary_cross_entropy(outputs, labels.float())
        return outputs, loss

    def configure_optimizers(self):   # ì•Œê³ ë¦¬ì¦˜ ì„¤ì •
        optimizer = torch.optim.Adam(self.parameters(), lr=0.001)
        return optimizer
    def training_step(self, batch, batch_idx):
        outputs, loss = self.process_batch(batch)
        self.log('train_loss', loss)
        return loss

    def test_step(self, batch, batch_idx):
        outputs, loss = self.process_batch(batch)
        images, labels = batch
        accuracy = self.accuracy(outputs, labels)
        self.log('test_accuracy', accuracy)
        return loss
```

```python copy
from torch.utils.data import Dataset

class BinaryMNIST(Dataset):
    def __init__(self, dataset, class_1, class_2):
        self.dataset = dataset
        self.indices = [i for i, (_, label) in enumerate(dataset) 
                        if label in [class_1, class_2]] # ì›í•˜ëŠ” í´ë˜ìŠ¤ì˜ ì¸ë±ìŠ¤ë§Œ
        self.class_map = {class_1: 0, class_2: 1} # class_1ì€ 0ìœ¼ë¡œ, class_2ëŠ” 1ë¡œ
        
    def __len__(self):
        return len(self.indices)
    
    def __getitem__(self, idx):
        image, label = self.dataset[self.indices[idx]] # idxë²ˆì§¸ ë°ì´í„°
        return image, torch.tensor(self.class_map[label]) # ë ˆì´ë¸” ë°”ê¾¸ê¸°
        # label.item() -> label

binary_train_dataset = BinaryMNIST(train_dataset, 3, 4)  # 2ë²ˆê³¼ 6ë²ˆì˜ ì´í•­ ë¶„ë¥˜
```
